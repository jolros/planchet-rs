1//! A Rust wrapper for the Numista API.
2//!
3//! # Examples
4//!
5//! ## Basic Search
6//!
7//! ```no_run
8//! use planchet::models::SearchTypesParams;
9//! use planchet::ClientBuilder;
10//!
11//! #[tokio::main]
12//! async fn main() {
13//!     let client = ClientBuilder::new()
14//!         .api_key("YOUR_API_KEY")
15//!         .build()
16//!         .unwrap();
17//!
18//!     let params = SearchTypesParams::new().q("victoria").year_range(1850, 1900);
19//!     let response = client.search_types(&params).await.unwrap();
20//!
21//!     println!("Found {} types", response.count);
22//! }
23//! ```
24//!
25//! ## Adding a Collected Item
26//!
27//! ```no_run
28//! use planchet::models::{AddCollectedItemParams, Grade};
29//! use planchet::ClientBuilder;
30//!
31//! #[tokio::main]
32//! async fn main() {
33//!     let client = ClientBuilder::new()
34//!         .api_key("YOUR_API_KEY")
35//!         .build()
36//!         .unwrap();
37//!
38//!     let item = AddCollectedItemParams::new(12345) // Type ID
39//!         .quantity(1)
40//!         .grade(Grade::Xf)
41//!         .private_comment("Bought at local coin show");
42//!
43//!     let result = client.add_collected_item(123, &item).await; // User ID
44//!
45//!     match result {
46//!         Ok(item) => println!("Added item with ID: {}", item.id),
47//!         Err(e) => eprintln!("Error adding item: {}", e),
48//!     }
49//! }
50//! ```
51//!
52//! ## Streaming and Error Handling
53//!
54//! This example demonstrates how to use the streaming API to fetch all results
55//! for a search and how to handle specific API errors.
56//!
57//! ```no_run
58//! use planchet::models::SearchTypesParams;
59//! use planchet::{ClientBuilder, Error};
60//! use futures::stream::TryStreamExt;
61//!
62//! #[tokio::main]
63//! async fn main() {
64//!     let client = ClientBuilder::new()
65//!         .api_key("YOUR_API_KEY")
66//!         .build()
67//!         .unwrap();
68//!
69//!     let params = SearchTypesParams::new().q("galleon");
70//!
71//!     let results = client.stream_all_types(params)
72//!         .try_collect::<Vec<_>>()
73//!         .await;
74//!
75//!     match results {
76//!         Ok(types) => {
77//!             println!("Successfully fetched {} types.", types.len());
78//!         }
79//!         Err(Error::ApiError(e)) if e.is_rate_limit_exceeded() => {
80//!             eprintln!("Rate limit exceeded. Please try again later.");
81//!         }
82//!         Err(e) => {
83//!             eprintln!("An unexpected error occurred: {}", e);
84//!         }
85//!     }
86//! }
87//! ```
