7// Centralized printing function.
8fn print_indented(msg: &str, indent: usize) {
9    println!("{:indent$}{}", "", msg, indent = indent);
10}
11
12fn print_key_value<T: std::fmt::Display>(key: &str, value: Option<T>, indent: usize) {
13    if let Some(v) = value {
14        print_indented(&format!("{}: {}", key.replace('_', " "), v), indent);
15    }
16}

18fn print_coin_side(label: &str, side: Option<&CoinSide>, indent: usize) {
19    if let Some(s) = side {
20        print_indented(&format!("{}:", label.replace('_', " ")), indent);
21        let next_indent = indent + 2;
22        if let Some(engravers) = &s.engravers {
23            if !engravers.is_empty() {
24                print_key_value("engravers", Some(engravers.join(", ")), next_indent);
25            }
26        }
27        if let Some(designers) = &s.designers {
28            if !designers.is_empty() {
29                print_key_value("designers", Some(designers.join(", ")), next_indent);
30            }
31        }
32        print_key_value("description", s.description.as_ref(), next_indent);
33        print_key_value("lettering", s.lettering.as_ref(), next_indent);
34        print_key_value("unabridged_legend", s.unabridged_legend.as_ref(), next_indent);
35        print_key_value("lettering_translation", s.lettering_translation.as_ref(), next_indent);
36        print_key_value("picture", s.picture.as_ref().map(Url::as_str), next_indent);
37    }
38}

40fn print_demonetization(label: &str, demonetization: Option<&Demonetization>, indent: usize) {
41    if let Some(d) = demonetization {
42        print_indented(&format!("{}:", label.replace('_', " ")), indent);
43        let next_indent = indent + 2;
44        print_key_value("is_demonetized", Some(d.is_demonetized), next_indent);
45        if let Some(date) = d.demonetization_date {
46            print_key_value("demonetization_date", Some(date.format("%Y-%m-%d").to_string()), next_indent);
47        }
48    }
49}

68fn print_ruling_authorities(label: &str, authorities: Option<&Vec<RulingAuthority>>, indent: usize) {

70        if !a.is_empty() {
71            print_indented(&format!("{}:", label.replace('_', " ")), indent);

119pub fn print_numista_type(type_: Option<&NumistaType>, indent: usize) {
120    if let Some(t) = type_ {
121        // Print the required fields first
122        print_key_value("id", Some(t.id), indent);
123        print_key_value("url", t.url.as_ref().map(|u| u.to_string()), indent);
124        print_key_value("title", Some(t.title.clone()), indent);
125        print_key_value("category", Some(t.category.to_string()), indent);
126
127        print_issuer("issuer", t.issuer.as_ref(), indent);
128        print_issuing_entity("issuing_entity", t.issuing_entity.as_ref(), indent);
129        print_issuing_entity(
130            "secondary_issuing_entity",
131            t.secondary_issuing_entity.as_ref(),
132            indent,
133        );
134        print_key_value("min_year", t.min_year, indent);
135        print_key_value("max_year", t.max_year, indent);
136        print_key_value("type_name", t.type_name.as_ref(), indent);
137
138        if let Some(v) = &t.value {
139            print_key_value("value", v.text.as_ref(), indent);
140        }
141
142        print_ruling_authorities("ruling_authorities", t.ruler.as_ref(), indent);
143        print_key_value("shape", t.shape.as_ref(), indent);
144
145        if let Some(c) = &t.composition {
146            print_key_value("composition", c.text.as_ref(), indent);
147        }
148
149        if let Some(tech) = &t.technique {
150            print_key_value("technique", tech.text.as_ref(), indent);
151        }
152
153        print_demonetization("demonetization", t.demonetization.as_ref(), indent);
154        print_key_value("weight", t.weight, indent);
155        print_key_value("size", t.size, indent);
156        print_key_value("size2", t.size2, indent);
157        print_key_value("thickness", t.thickness, indent);
158
159        print_coin_side("obverse", t.obverse.as_ref(), indent);
160        print_coin_side("reverse", t.reverse.as_ref(), indent);
161        print_coin_side("edge", t.edge.as_ref(), indent);
162        print_coin_side("watermark", t.watermark.as_ref(), indent);
163
164        if let Some(mints) = &t.mints {
165            if !mints.is_empty() {
166                let mint_names: Vec<String> = mints.iter().map(|m| m.name.clone()).collect();
167                print_key_value("mints", Some(mint_names.join(", ")), indent);
168            }
169        }
170
171        print_printers("printers", t.printers.as_ref(), indent);
172        print_key_value("series", t.series.as_ref(), indent);
173        print_key_value("commemorated_topic", t.commemorated_topic.as_ref(), indent);
174        print_key_value("comments", t.comments.as_ref(), indent);
175
176        if let Some(tags) = &t.tags {
177            if !tags.is_empty() {
178                print_key_value("tags", Some(tags.join(", ")), indent);
179            }
180        }
181
182        print_related_types("related_types", t.related_types.as_ref(), indent);
183        print_references("references", t.references.as_ref(), indent);
184    }
